-- Gestores
CREATE TABLE gestores (
id SERIAL PRIMARY KEY,
nome TEXT NOT NULL,
email TEXT UNIQUE NOT NULL,
telefone TEXT,
senha TEXT NOT NULL
)

-- Produtos
-- 1️⃣ Criação dos tipos ENUM
CREATE TYPE tamanho_enum AS ENUM ('Pequeno', 'Médio', 'Grande');
CREATE TYPE rotulo_enum AS ENUM ('Sem rótulo', 'Padrão', 'Personalizado');
CREATE TYPE tipo_embalagem_enum AS ENUM ('Vidro', 'Plástico', 'Acrílico');
CREATE TYPE cor_tampa_enum AS ENUM ('Verde', 'Laranja', 'Roxo');
CREATE TYPE acabamento_superficie_enum AS ENUM ('Fosco', 'Brilhante', 'Texturizado');

-- 2️⃣ Criação da tabela produtos
CREATE TABLE produtos (
    id SERIAL PRIMARY KEY,
    nome TEXT NOT NULL,
    preco NUMERIC(10, 2) NOT NULL,
    descricao TEXT,
    imagem TEXT, -- pode ser uma URL ou base64

    tamanho tamanho_enum NOT NULL,
    rotulo rotulo_enum NOT NULL,
    tipo_embalagem tipo_embalagem_enum NOT NULL,
    cor_tampa cor_tampa_enum NOT NULL,
    acabamento_superficie acabamento_superficie_enum NOT NULL
);


-- Pedidos
CREATE TABLE pedidos (
    id SERIAL PRIMARY KEY,
    codigo_pedido TEXT UNIQUE NOT NULL, -- visível ao usuário
    usuario_id INTEGER REFERENCES usuarios(id),
    data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    status TEXT CHECK (status IN (
        'Pedido realizado',
        'Pedido em preparo',
        'Pedido enviado',
        'Pedido entregue',
        'Pedido cancelado'
    )) NOT NULL,
    valor_total NUMERIC(10, 2) NOT NULL
);
